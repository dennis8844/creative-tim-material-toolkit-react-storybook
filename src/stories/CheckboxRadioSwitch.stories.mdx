import classNames from "classnames";
import { Meta, Canvas, Story } from '@storybook/addon-docs/blocks';
import { makeStyles } from "@material-ui/core/styles";
import {
    Checkbox,
    Radio,
    Switch,
    FormControlLabel
} from "@material-ui/core";
import {
    Check,
    FiberManualRecord
} from "@material-ui/icons";

import customCheckboxRadioSwitchStyles from "../_material-toolkit-components/_assets/jss/material-kit-react/customCheckboxRadioSwitch.js";

import * as boolComponentStories from './_storyData/boolComponentStories.js';
const argTypes = boolComponentStories.defaultArgTypes;
const useStyles = makeStyles(customCheckboxRadioSwitchStyles);
//shared vars
const classes = makeStyles(customCheckboxRadioSwitchStyles);
//checkbox vars
const [checked, setChecked] = React.useState([24, 22]);
const checkboxWrapperDiv = classNames(
    classes.checkboxAndRadio,
    classes.checkboxAndRadioHorizontal
);
const handleToggle = value => {
const currentIndex = checked.indexOf(value);
const newChecked = [...checked];
if (currentIndex === -1) {
  newChecked.push(value);
} else {
  newChecked.splice(currentIndex, 1);
}
setChecked(newChecked);
};
//radio vars
const [selectedEnabled, setSelectedEnabled] = React.useState("b");
const radioWrapperDiv = classNames(
    classes.checkboxAndRadio,
    classes.checkboxAndRadioHorizontal
);
//select vars
const [checkedA, setCheckedA] = React.useState(true);
const [checkedB, setCheckedB] = React.useState(false);

<Meta title="Components/CheckboxRadioSwitch" components={Checkbox, Radio, Switch} argTypes={argTypes} />

# Checkboxes, Radios and Switches

We haven't extended these components from [Material-UI](https://material-ui-next.com/demos/selection-controls/?ref=creativetim). We've just added our style over their components.

## Styles

You will find the styles for this component in
`src/_assets/jss/material-kit-react/customCheckboxRadioSwitch.js`.

Let's take a look at each one of them, and how to use our styles.

## Checkboxes

<Canvas>
  <Story name="Checkboxes">
    <div>
      <div className={checkboxWrapperDiv}>
        <FormControlLabel
          control={
            <Checkbox
              tabIndex={-1}
              onClick={() => handleToggle(21)}
              checkedIcon={<Check className={classes.checkedIcon} />}
              icon={<Check className={classes.uncheckedIcon} />}

          }
          classes={{
            label: classes.label,
            disabled: classes.disabledCheckboxAndRadio
          }}
          label="Disabled Unchecked"
        />
      </div>
      <div className={wrapperDiv}>
        <FormControlLabel
          disabled
          control={
            <Checkbox
              tabIndex={-1}

                checked.indexOf(24) !== -1 ? true : false
              }

              icon={<Check className={classes.uncheckedIcon} />}
              classes={{ checked: classes.checked }}
            />
          }
          classes={{
            label: classes.label,
            disabled: classes.disabledCheckboxAndRadio
          }}
          label="Disabled Checked"
        />
      </div>
    </div>
  </Story>
</Canvas>
